            <button class="btn btn-export" id="exportBtn">Export Masked Image</button>
            <div class="mask-info" id="maskInfo"></div>

    <script>
        // Canvas and context
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');
        
        // DOM elements
        const imageUpload = document.getElementById('imageUpload');
        const canvasContainer = document.getElementById('canvasContainer');
        const controls = document.getElementById('controls');
        const instructions = document.getElementById('instructions');
        const autoMaskBtn = document.getElementById('autoMaskBtn');
        const clearBtn = document.getElementById('clearBtn');
        const exportBtn = document.getElementById('exportBtn');
        const maskInfo = document.getElementById('maskInfo');

        // State
        let uploadedImage = null;
        let isDrawing = false;
        let startX, startY;
        let maskRect = null;

        // Handle image upload
        imageUpload.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(event) {
                const img = new Image();
                img.onload = function() {
                    uploadedImage = img;
                    
                    // Set canvas size to match image
                    const maxWidth = 800;
                    const scale = Math.min(1, maxWidth / img.width);
                    canvas.width = img.width * scale;
                    canvas.height = img.height * scale;
                    
                    // Draw image
                    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                    
                    // Show canvas and controls
                    canvasContainer.style.display = 'block';
                    controls.style.display = 'flex';
                    instructions.style.display = 'block';
                    
                    // Clear any previous mask
                    maskRect = null;
                    updateMaskInfo();
                };
                img.src = event.target.result;
            };
            reader.readAsDataURL(file);
        });

        // Manual masking - Mouse events
        canvas.addEventListener('mousedown', function(e) {
            const rect = canvas.getBoundingClientRect();
            startX = e.clientX - rect.left;
            startY = e.clientY - rect.top;
            isDrawing = true;
        });

        canvas.addEventListener('mousemove', function(e) {
            if (!isDrawing) return;
            
            const rect = canvas.getBoundingClientRect();
            const currentX = e.clientX - rect.left;
            const currentY = e.clientY - rect.top;
            
            // Redraw image
            ctx.drawImage(uploadedImage, 0, 0, canvas.width, canvas.height);
            
            // Draw rectangle
            const width = currentX - startX;
            const height = currentY - startY;
            
            ctx.fillStyle = 'rgba(0, 0, 0, 0.7)';
            ctx.fillRect(startX, startY, width, height);
            
            // Draw border
            ctx.strokeStyle = '#3b82f6';
            ctx.lineWidth = 2;
            ctx.strokeRect(startX, startY, width, height);
        });

        canvas.addEventListener('mouseup', function(e) {
            if (!isDrawing) return;
            
            const rect = canvas.getBoundingClientRect();
            const endX = e.clientX - rect.left;
            const endY = e.clientY - rect.top;
            
            maskRect = {
                x: Math.min(startX, endX),
                y: Math.min(startY, endY),
                width: Math.abs(endX - startX),
                height: Math.abs(endY - startY)
            };
            
            isDrawing = false;
            redrawWithMask();
            updateMaskInfo();
        });

        // Auto mask eyes (simple heuristic - upper-middle portion of face)
        autoMaskBtn.addEventListener('click', function() {
            if (!uploadedImage) return;
            
            // Simple automatic mask - assumes face is centered
            // Eyes typically in upper 30-50% of image, centered horizontally
            const eyeRegionWidth = canvas.width * 0.6;
            const eyeRegionHeight = canvas.height * 0.15;
            const x = (canvas.width - eyeRegionWidth) / 2;
            const y = canvas.height * 0.25;
            
            maskRect = {
                x: x,
                y: y,
                width: eyeRegionWidth,
                height: eyeRegionHeight
            };
            
            redrawWithMask();
            updateMaskInfo();
        });

        // Clear mask
        clearBtn.addEventListener('click', function() {
            if (!uploadedImage) return;
            
            maskRect = null;
            ctx.drawImage(uploadedImage, 0, 0, canvas.width, canvas.height);
            updateMaskInfo();
        });

        // Export masked image
        exportBtn.addEventListener('click', function() {
            if (!uploadedImage) return;
            
            // Create temporary canvas for export
            const exportCanvas = document.createElement('canvas');
            exportCanvas.width = canvas.width;
            exportCanvas.height = canvas.height;
            const exportCtx = exportCanvas.getContext('2d');
            
            // Draw image
            exportCtx.drawImage(uploadedImage, 0, 0, canvas.width, canvas.height);
            
            // Draw mask if exists
            if (maskRect) {
                exportCtx.fillStyle = 'rgba(0, 0, 0, 0.9)';
                exportCtx.fillRect(maskRect.x, maskRect.y, maskRect.width, maskRect.height);
            }
            
            // Export as PNG
            exportCanvas.toBlob(function(blob) {
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = 'masked-image.png';
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
                
                maskInfo.textContent = '‚úì Image exported successfully!';
                setTimeout(() => updateMaskInfo(), 2000);
            });
        });

        // Helper function to redraw with mask
        function redrawWithMask() {
            if (!uploadedImage) return;
            
            ctx.drawImage(uploadedImage, 0, 0, canvas.width, canvas.height);
            
            if (maskRect) {
                ctx.fillStyle = 'rgba(0, 0, 0, 0.7)';
                ctx.fillRect(maskRect.x, maskRect.y, maskRect.width, maskRect.height);
                
                ctx.strokeStyle = '#3b82f6';
                ctx.lineWidth = 2;
                ctx.strokeRect(maskRect.x, maskRect.y, maskRect.width, maskRect.height);
            }
        }

        // Update mask info display
        function updateMaskInfo() {
            if (maskRect) {
                maskInfo.textContent = `Mask applied: ${Math.round(maskRect.width)}√ó${Math.round(maskRect.height)}px`;
            } else {
                maskInfo.textContent = 'No mask applied';
            }
        }
    </script>
